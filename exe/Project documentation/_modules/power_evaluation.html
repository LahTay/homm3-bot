    """

[docs]def creatures_power(slots):
    """
    function evaluates int: strength of Class Slots

    :param slots: Slots object
    :return: int - power
    """
    power = 0
    for s in slots.slots:
        power = power + creature_power(s)

    return power

[docs]def compare_power(x1, x2):
    """
    Compares power between x1 (int) and x2(int)

    :param x1: Strength parameter for first instance
    :param x2: Strength parameter for second instance
    :return: (float) ratio of strength. returns 1 -> win 100%. Returns -10 -> win 1%
    """
    ratio = x2/x1   
    if ratio < 0.5 or x2 == 0:
        print("win 100%")
        return 1
    elif ratio < 0.6:
        print("win 95%")
        return 0.9
    elif ratio < 0.7:
        print("win 90%")
        return 0.75
    elif ratio < 0.8:
        print("win 80%")
        return 0.2
    elif ratio < 0.9:
        print("win 65%")
        return -0.1
    elif ratio < 1:
        print("win 50% - our bot inteligence")
        return -0.3
    elif ratio < 1.1:
        print("win 10%")
        return -0.5
    elif ratio < 1.2:
        print("win 10%")
        return -1
    else:
        print("win 1%")
        return -10
